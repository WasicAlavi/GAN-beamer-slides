\section{Advantages and Limitations}

\begin{frame}{Advantages}
\begin{itemize}
\item Sharp realistic samples.
\item No explicit likelihood required.
\item Fast sampling via forward pass.
\end{itemize}
\end{frame}

\begin{frame}{Limitations}
\begin{itemize}
\item Training instability.
\item Mode collapse.
\item Hyperparameter sensitivity.
\end{itemize}
\end{frame}
