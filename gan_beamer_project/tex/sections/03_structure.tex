\section{Structure of GAN}

\begin{frame}{Generator and Discriminator}
\begin{block}{Generator}
Input: random noise \\
Output: synthetic data
\end{block}

\begin{block}{Discriminator}
Input: real or fake sample \\
Output: probability of being real
\end{block}
\end{frame}

\begin{frame}{GAN Architecture Diagram}
\centering
\begin{tikzpicture}[node distance=1.5cm,
  block/.style={draw, rounded corners, align=center, minimum width=2.5cm},

]

\node[block, right=of noise] (gen) {Generator};
\node[block, right=of gen] (fake) {Fake Sample};

\draw[arrow] (real) -- (disc);
\draw[arrow] (disc) -- (out);

\end{tikzpicture}
\end{frame}
